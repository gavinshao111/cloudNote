    ~GMqttClient() {cout << "GMqttClient desstruct" << endl;}
private:

};
class A {
public:
    A(GMqttClient& mqtt) : m_mqtt(mqtt) {cout << "A construct" << endl;}
    ~A() {cout << "A desstruct" << endl;}

    GMqttClient& m_mqtt;
            
};

构造函数初始化列表执行顺序：
对成员变量的初始化，是严格按照声明次序，而不是在初始化列表中的顺序进行初始化

链接选项 -static 的作用：
在某库同时存在动态库和静态库的时候，默认链接动态库，加上该选项会链接静态库。
